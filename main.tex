\question{计算不定积分 $\int (4x^2 - \frac{3}{4}x + 1) \, dx$。}

\begin{solution}
    根据积分的线性性质和幂函数的积分公式，有：
    $$
    \int \left(4x^2 - \frac{3}{4}x + 1\right) \, dx = \frac{4}{3}x^3 - \frac{3}{8}x^2 + x + C
    $$
    其中 $C$ 是积分常数。
\end{solution}

\question{考虑事件 $A$ 和 $B$，$P(A)=0.5$，$P(B)=0.6$，且 $P(A \cap B)=0.3$。}
    \begin{enumerate}[(a)]
        \item 计算 $P(A \cup B)$。
        \item 计算 $P(A^c)$。
    \end{enumerate}

\begin{solution}
    \begin{enumerate}[(a)]
        \item $P(A \cup B) = P(A) + P(B) - P(A \cap B) = 0.5 + 0.6 - 0.3 = \mathbf{0.8}$
        \item $P(A^c) = 1 - P(A) = 1 - 0.5 = \mathbf{0.5}$
    \end{enumerate}
\end{solution}
